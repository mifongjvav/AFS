function 函数_1(x, y, 百分比, 精度, 文字, i) {
  定义脚本变量("脚本变量_大小");
  脚本变量_大小 = 50;
  if (i === 0) {
    移到位置(x, y + 脚本变量_大小);
    落笔();
    if (文字 === "显示进度") {
      函数_1(x, y, 百分比, 精度, "显示进度", i + 1);
    } else {
      函数_1(x, y, 百分比, 精度, "隐藏进度", i + 1);
    }
  } else {
    if (i <= 百分比 / 100 * 360 / 精度) {
      移到位置(x + 三角函数("sin", i * 精度) * 脚本变量_大小, y + 三角函数("cos", i * 精度) * 脚本变量_大小);
      if (文字 === "显示进度") {
        函数_1(x, y, 百分比, 精度, "显示进度", i + 1);
      } else {
        函数_1(x, y, 百分比, 精度, "隐藏进度", i + 1);
      }
    } else {
      抬笔();
      if (文字 === "显示进度") {
        移到位置(x, y);
        设置文字印章(连接字符串(百分比, "%"), 24, "居中");
      }
    }
  }
}
当开始运行(() => {
  设置画笔粗细("10");
  遍历循环(0, 100, 1, i => {
    清除画笔();
    设置画笔颜色("#D8D8D8");
    函数_1(0, 0, 100, 5, "隐藏进度", 0);
    设置画笔颜色("#000000");
    函数_1(0, 0, i, 5, "显示进度", 0);
  });
});
