function 函数_2(x, y, 百分比, 长度) {
  移到位置(x - 长度 / 2, 0);
  设置画笔颜色("#D8D8D8");
  落笔();
  移到位置(x + 长度 / 2, y);
  抬笔();
  移到位置(x - 长度 / 2, 0);
  设置画笔颜色("#000000");
  落笔();
  移到位置(x - 长度 / 2 + 长度 * (百分比 / 100), y);
  抬笔();
  移到位置(x, y - 30);
  设置文字印章(连接字符串(百分比, "%"), 24, "居中");
}
函数_2(0, 0, 10, 100);
