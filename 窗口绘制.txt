function 函数_3(x, y, w, h, name, title) {
  if (判断列表值是否存在("列表_TKS", name)) {
    函数返回();
  } else {
    添加列表项(name, "列表_TKS");
  }
  定义脚本变量("脚本变量_line");
  设置画笔颜色("#39c5bb");
  设置画笔粗细("1");
  抬笔();
  移到位置(x, y);
  落笔();
  移到位置(角色坐标("自己", "x坐标") + w, 角色坐标("自己", "y坐标"));
  移到位置(角色坐标("自己", "x坐标"), 角色坐标("自己", "y坐标") - 24);
  移到位置(角色坐标("自己", "x坐标") - w, 角色坐标("自己", "y坐标"));
  移到位置(角色坐标("自己", "x坐标"), 角色坐标("自己", "y坐标") + 24);
  抬笔();
  设置画笔粗细("1");
  重复循环(2, () => {
    移到位置(x, y);
    脚本变量_line = 0;
    一步执行(() => {
      重复直到(脚本变量_line > 24, () => {
        落笔();
        移到位置(角色坐标("自己", "x坐标") + w, 角色坐标("自己", "y坐标"));
        抬笔();
        移到位置(x, 角色坐标("自己", "y坐标") - 1);
        改变变量("脚本变量_line", "增加", 1);
      });
    });
  });
  移到位置(角色坐标("自己", "x坐标") + 4, 角色坐标("自己", "y坐标") + 12);
  设置画笔颜色("#ffffff");
  if (12 * 字符串长度(title) > w) {
    设置文字印章(连接字符串(选择字符串(title, 1, 5), "......"), 18, "居右");
  } else {
    设置文字印章(title, 18, "居右");
  }
  抬笔();
  移到位置(x, y);
  变量__COMM_00 = w;
  设置画笔颜色("#ffffff");
  抬笔();
  移到位置(x, y);
  改变y坐标("减少", 24);
  落笔();
  移到位置(角色坐标("自己", "x坐标") + w, 角色坐标("自己", "y坐标"));
  移到位置(角色坐标("自己", "x坐标"), 角色坐标("自己", "y坐标") - h);
  移到位置(角色坐标("自己", "x坐标") - w, 角色坐标("自己", "y坐标"));
  移到位置(角色坐标("自己", "x坐标"), 角色坐标("自己", "y坐标") + h);
  抬笔();
  设置画笔粗细("1");
  重复循环(2, () => {
    移到位置(x, y);
    改变y坐标("减少", 24);
    脚本变量_line = 0;
    一步执行(() => {
      重复直到(脚本变量_line > h, () => {
        落笔();
        移到位置(角色坐标("自己", "x坐标") + w, 角色坐标("自己", "y坐标"));
        抬笔();
        移到位置(x, 角色坐标("自己", "y坐标") - 1);
        改变变量("脚本变量_line", "增加", 1);
      });
    });
  });
}
