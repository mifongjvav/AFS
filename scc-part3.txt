function 函数_2(x坐标, y坐标, 血条长度, 血条粗细, 当前血量, 血量总值) {
  移到位置(x坐标, y坐标);
  设置面向角度("0");
  移动步数(科学计算("-", 血条长度 / 2));
  设置画笔粗细(血条粗细);
  设置画笔颜色("#000000");
  落笔();
  移动步数(血条长度);
  抬笔();
  移动步数(科学计算("-", 血条长度));
  设置画笔粗细(血条粗细 - 7);
  设置画笔颜色("#ffffff");
  落笔();
  移动步数(血条长度);
  抬笔();
  移动步数(科学计算("-", 血条长度));
  if (当前血量 / 血量总值 >= 0.75) {
    设置画笔颜色("#37E547");
  } else if (当前血量 / 血量总值 >= 0.5 && 当前血量 / 血量总值 < 0.75) {
    设置画笔颜色("#FADB14");
  } else if (当前血量 / 血量总值 >= 0.25 && 当前血量 / 血量总值 < 0.5) {
    设置画笔颜色("#FFB029");
  } else {
    设置画笔颜色("#E44916");
  }
  if (当前血量 <= 血量总值 && 当前血量 > 0) {
    落笔();
    移动步数(血条长度 / 血量总值 * 当前血量);
  } else if (当前血量 <= 0) {} else {
    落笔();
    移动步数(血条长度);
  }
  抬笔();
  if (字符串包含判断("对象指示", "↓")) {
    移到位置(x坐标, y坐标);
    设置面向角度("0");
    设置画笔粗细("3");
    设置画笔颜色("#000000");
    落笔();
    if (字符串包含判断("指示尺寸（例：20步）", "↓")) {}
    移动步数(20 / 2);
    旋转角度(-120);
    移动步数(20);
    旋转角度(-120);
    移动步数(20);
    旋转角度(-120);
    移动步数(20);
    抬笔();
  }
}
函数_2(0, 0, 0, 0, 0, 0);
